import { AppsList, getStaticProps as gsp } from '../../components/AppsList';
import { useSSG } from 'nextra/ssg';

export const getStaticProps = gsp;

export const Apps = () => {
  // Get the data from SSG, and render it as a component.
  const { apps } = useSSG();
  return <AppsList apps={apps} />;
};
